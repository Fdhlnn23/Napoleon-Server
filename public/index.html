Contoh Penggunaan di Script
-- Roblox Script Kamu
local HttpService = game:GetService("HttpService")

local API_URL = "https://YOUR_PROJECT.vercel.app/api/check"
local key = "MASUKKAN_KEY_DISINI"
local hwid = game.Players.LocalPlayer.UserId

local result = HttpService:RequestAsync({
    Url = API_URL .. "?key=" .. key .. "&hwid=" .. hwid,
    Method = "GET"
})

local data = HttpService:JSONDecode(result.Body)

if data.valid then
    print("✅ Key valid! Selamat datang.")
    -- Lanjutkan script kamu di sini
else
    warn("❌ " .. data.message)
    -- Hentikan script
end
